import { Head } from 'ice';

  return (
    <div>
      <Head>
        <title>Hello Home</title>
      </Head>
      <div>Home</div>
    </div>
  );
};


export default Home;

# 页面滚动

scrollToTop：用于渲染页面前是否需要将当前页面滚动至顶部，这个配置用于嵌套路由的应用场景。

Type: Boolean (default: false)

const HomePage = () => {
  return (
    // jsx code...
  )
}

HomePage.pageConfig = {
  scrollToTop: true
}
